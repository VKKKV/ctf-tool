26 static int can_use_lsb(info_s * infos)
27 {
28  assert(infos);
29  /* Si le fichier hôte est un fichier BMP non compressé. Si le nombre de bits
30  * utilisés pour coder la couleur de chaque pixel est inférieur ou égal à 8
31  * (soit 256 couleurs), le deuxième en-tête est suivi d’une table des
32  * couleurs utilisées dans l’image => LSB non proposé */
33  if (infos->host.type == BMP_UNCOMPRESSED && infos->host.file_info.bmp.pixel_length > 8) {
34  // calcul du nombre d'octets représentant l'image brute
35  uint64_t nb_bits_pic =
36  ((infos->host.file_info.bmp.pixel_length) * (infos->host.file_info.bmp.pixel_number)) /
37  8;
38  // calcul du nombre de bits modifiables pour l'algorithme LSB
39  nb_bits_pic /= 4;
40 
41  // si la taille du fichier a cacher est ok -> LSB possible
42  if ((infos->hidden_length * 8) <= nb_bits_pic)
43  return 0;
44  }
45 
46  /* Si le fichier hôte est un fichier WAVE-PCM. */
47  else if (infos->host.type == WAV_PCM) {
48  /* Calcul du nombre de bits modifiables pour l'algorithme LSB :
49  ([nb d'octets de data] / ([nb de bits par sample] / 8)) = [nb de bit de poids faible d'audio]
50  [nb de bit de poids faible d'audio] * 2 = [nb de bits modifiable]
51  (car on modifie les 2 bits de poids faible par sample). */
52  uint64_t nb_bits_modif =
53  ((infos->host.file_info.wav.data_size) / (infos->host.file_info.wav.chunk_size / 8)) *
54  2;
55 
56  /* Si le nombre de bit du fichier à cacher est inférieur ou égal au
57  * nombre de bit modifiables => LSB utilisable. */
58  if ((infos->hidden_length * 8) <= nb_bits_modif)
59  return 0;
60  }
61  /* Si le fichier hote est un fichier MP3. */
62  else if (infos->host.type == MP3 &&
63  (infos->host.file_info.mp3.fr_nb * MP3_HDR_NB_BITS_MODIF >= infos->hidden_length * 8))
64  return 0;
65  /* Sinon, on ne peux pas utiliser LSB. */
66  return 1;
67 }

145  if ((infos->host.type == BMP_COMPRESSED) || (infos->host.type == BMP_UNCOMPRESSED)) {
146  uint32_t begin_pic, length_pic, pixel_width, pixel_height;
147  uint16_t pixel_length;

151  return 1;

153  return 1;

172  if (fseek(infos->host.host, BMP_DEF_NB_PIXEL, SEEK_SET) == -1)
173  return 1;

175  return 1;

178  if (fread(&pixel_height, sizeof(uint32_t), 1, infos->host.host) != 1)
179  return 1;
180  infos->host.file_info.bmp.pixel_number = pixel_width * pixel_height;
181  return 0;

184  // http://www.libpng.org/pub/png/spec/1.2/PNG-Chunks.html
185  else if (infos->host.type == PNG) {

187  if (fseek(infos->host.host, PNG_DEF_IHDR, SEEK_SET) == -1)
188  return 1;

191  return 1;

198  if (fseek(infos->host.host, LENGTH_SIG_PNG, SEEK_SET))

204  return perror("PNG file: Can't read ID of chunk"), 1;
205  // on cherche le chunk IEND pour connaitre la taille du fichier
206  while (chunk_id != SIG_IEND) {
207  if (fseek(infos->host.host, chunk_size + LENGTH_CRC, SEEK_CUR))
208  return perror("PNG file: Can not move in the file"), 1;
209  if (fread(&chunk_size, sizeof(uint32_t), 1, infos->host.host) != 1)
210  return perror("PNG file: Can't read size of chunk"), 1;
211  chunk_size = stegx_be32toh(chunk_size);
212  if (fread(&chunk_id, sizeof(uint32_t), 1, infos->host.host) != 1)
213  return perror("PNG file: Can't read ID of chunk"), 1;
214  }
215  uint32_t file_length = ftell(infos->host.host) + LENGTH_IEND;
216  infos->host.file_info.png.data_size = file_length - infos->host.file_info.png.header_size;
217  return 0;
218  }

235  return perror("WAVE file: Can't read the size of subchunk"), 1;

332  * été lu => on à lu 4 octets de données en trop (exemple, la signature). */
333  if (!mp3_id3v1_hdr_test(hdr) && !feof(h))
334  infos->host.file_info.mp3.eof -= sizeof(hdr);
335  return 0;
336  }
337 
338  /* Structure AVI : structure vide. */
339  else if (infos->host.type == AVI_COMPRESSED || infos->host.type == AVI_UNCOMPRESSED)
340  return 0;
341  /* Format non reconnu => erreur. */
342  else
343  return 1;

346 int stegx_suggest_algo(info_s * infos)
347 {
348  /* Test si on est en mode insertion, si oui, remplit la structure

353  // Lecture de la taille du fichier à cacher.
354  if (fseek(infos->hidden, 0, SEEK_END))
355  return stegx_errno = ERR_SUGG_ALGOS, perror("Can't move to the end of hidden file"), 1;
356  uint64_t read_hidden_length = ftell(infos->hidden);
357  if (read_hidden_length == 0)
358  return stegx_errno = ERR_HIDDEN_FILE_EMPTY, 1;
359  // Précaution overflow.

365  /* Remplissage du tableau stegx_propos_algos pour savoir

377 int stegx_choose_algo(info_s * infos, algo_e algo_choosen)
378 {
379  if (infos->mode == STEGX_MODE_EXTRACT)
380  return stegx_errno = ERR_SUGG_ALGOS, 1;
381  /* Si l'utilisateur n'a pas choisi de mot de passe, on en crée un par défaut aléatoirement. */
382  if (infos->method == STEGX_WITHOUT_PASSWD) {
383  stegx_srand(time(NULL));
384  free(infos->passwd);
385  if (!(infos->passwd = calloc((LENGTH_DEFAULT_PASSWD + 1), sizeof(char))))
386  return perror("Can't allocate memory for password string"), 1;
387  // Génération de symboles ASCII >= 32 et <= 126.
388  for (int i = 0; i < LENGTH_DEFAULT_PASSWD; i++)
389  infos->passwd[i] = 32 + (stegx_rand() % 95);
390  }
391

393  /* Test que l'algorithme choisis à bien été proposé comme étant possible. Si
394  * oui, alors il est sauvegardé. Sinon, on lève une erreur. */
395  if (stegx_propos_algos[algo_choosen])
396  infos->algo = algo_choosen;
397  else
398  return stegx_errno = ERR_CHOICE_ALGO, 1;
